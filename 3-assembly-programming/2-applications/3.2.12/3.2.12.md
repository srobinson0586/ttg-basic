# Demonstrate understanding of bit shifting and masking

- Complete the code to ensure the program exits normally (ie with an error code of 0).
  - You must perform the following bitwise operations:
    - Bit shifting
    - Bit masking

```assembly
section .text
global _start
_start:
   mov rdi, 1
   mov rax, 0x34120000
   mov rbx, 0xFFAB3412
   ___ rax, __________
   ___ rbx, __________
   cmp eax, ebx
   je .myLabel
   jmp .badLabel

.myLabel:
   mov rdi, 0

   cmp ax, 0x3412
   jne .badLabel

   cmp bx, 0x3412
   jne .badLabel
   jmp .exit

.badLabel:
   mov rdi, 1

.exit:
   mov rax, 60
   syscall
```
